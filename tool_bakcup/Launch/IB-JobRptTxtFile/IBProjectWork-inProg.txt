SELECT 
	PCODE, PTITLE,CUSTOMER_DP.CUSTNAME,PRECEIVEDDATE,pnoofpages,pcno_2009,stypename,PRECEIVEDDATE, PDUEDATE, PCOMPLETEDDATE,
	PINVOICEDDATE,null as CATSDueDATE,PROJECTNO FROM PROJECTS_DP X
LEFT JOIN 
	CUSTOMER_DP 
ON 	X.CUSTNO=CUSTOMER_DP.CUSTNO 
LEFT JOIN 
	STYPE_DP 
ON 
	X.STYPENO=STYPE_DP.STYPENO 
where 
 PRECEIVEDDATE BETWEEN ':STARTDATE 00:00:00' AND ':ENDDATE 23:59:59' 
 AND X.CUSTNO in (:CUSTNO)
 AND (PINVOICEDDATE> ':ENDDATE' or PINVOICEDDATE is null)
Order by 
	PRECEIVEDDATE
