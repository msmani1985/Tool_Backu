SELECT 
	CUSTOMER_DP.CUSTNAME, PCODE, PTITLE,pnoofpages,stypename, PRECEIVEDDATE,PDUEDATE, PCOMPLETEDDATE,
	PINVOICEDDATE,	null as CATSDueDate,PROJECTNO,X.CUSTNO, PRINTNO, PPRODEDNO, PPRODMANNO, PADDITEMS, 
	PADDCHARGES,PINVOICED, INVNO, PDESCRIPTION, PCOMMENTS, PDIGITAL, X.DIGITALPRODNO, X.TPLATNO,FINSITENO,
	X.CONNO, STYPE_DP.STYPENAME, STYPE_DP.STYPENAME,CONTACT_DP.CONFIRSTNAME,DIGITALPRODUCTS_DP.PRODCODE, 
	TYPESETPLATFORM_DP.TPLATCODE, X.PFINALDESPATCHED 
FROM  PROJECTS_DP X 
JOIN  CUSTOMER_DP  ON  X.CUSTNO=CUSTOMER_DP.CUSTNO 
LEFT JOIN  DIGITALPRODUCTS_DP  ON  X.DIGITALPRODNO=DIGITALPRODUCTS_DP.DIGITALPRODNO 
LEFT JOIN  TYPESETPLATFORM_DP  ON  X.TPLATNO=TYPESETPLATFORM_DP.TPLATNO 
JOIN  STYPE_DP  ON  X.STYPENO=STYPE_DP.STYPENO 
LEFT JOIN  CONTACT_DP  ON  X.CONNO=CONTACT_DP.CONNO  
/*LEFT JOIN  plogevents_DP  ON X.projectno=plogevents_DP.projectno */ 
WHERE X.pdespatched='Y' and (PCOMPLETEDDATE BETWEEN ':STARTDATE 00:00:00' AND ':ENDDATE 23:59:59')
AND X.CUSTNO in (:CUSTNO)
ORDER BY PCOMPLETEDDATE
